CREATE TABLE solicitudes_cursos (
    id SERIAL PRIMARY KEY,
    titulo_propuesto VARCHAR(200) NOT NULL,
    categoria_id INTEGER REFERENCES categorias_cursos(id),
    solicitante_id INTEGER REFERENCES usuarios(id),
    justificacion_mercado TEXT,
    publico_objetivo TEXT,
    objetivos_generales TEXT,
    duracion_estimada INTEGER,
    prioridad VARCHAR(20) DEFAULT 'media', -- baja, media, alta
    estado VARCHAR(20) DEFAULT 'pendiente', -- pendiente, aprobada, rechazada
    comentarios_aprobacion TEXT,
    aprobado_por INTEGER REFERENCES usuarios(id),
    fecha_aprobacion TIMESTAMP,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

3.1. Asignación de Recursos

CREATE TABLE tareas_desarrollo (
    id SERIAL PRIMARY KEY,
    curso_id INTEGER REFERENCES cursos(id) ON DELETE CASCADE,
    modulo_id INTEGER REFERENCES modulos(id) ON DELETE CASCADE,
    leccion_id INTEGER REFERENCES lecciones(id) ON DELETE CASCADE,
    
    tipo_contenido VARCHAR(50), -- video, texto, quiz, recurso, etc.
    descripcion_tarea TEXT,
    asignado_a INTEGER REFERENCES usuarios(id),
    
    -- Especificaciones
    formato_requerido VARCHAR(50),
    duracion_estimada VARCHAR(50), -- para videos
    calidad_minima VARCHAR(20),
    tamano_maximo INTEGER, -- en MB
    
    fecha_limite TIMESTAMP,
    estado VARCHAR(20) DEFAULT 'pendiente',
    fecha_completacion TIMESTAMP,
    
    -- Control de calidad
    revisado_por INTEGER REFERENCES usuarios(id),
    comentarios_revision TEXT,
    requiere_correccion BOOLEAN DEFAULT false
);
3.2. Flujo de Desarrollo por Tipo de Contenido
A. CONTENIDO MULTIMEDIA (Videos)
text
Grabación → Edición → Compresión → Subtitulado → Optimización → Upload
Especificaciones Técnicas:

Resolución: 720p (máximo) para ahorro de datos

Formato: MP4 con H.264

Duración: 5-15 minutos (microlearning)

Tamaño: <50MB por video

Subtítulos: Obligatorios para accesibilidad

B. CONTENIDO TEXTUAL
text
Redacción → Revisión → Formateo HTML → Inserción imágenes → Prueba lectura
Estándares:

Longitud: 500-2000 palabras por lección

Formato: HTML semántico

Imágenes: Comprimidas, <100KB cada una

Lectura: Nivel de dificultad apropiado

Tipos de Preguntas:

 PANEL DE CONTROL DEL PROCESO
Vista Coordinador Académico:
sql
-- Query para dashboard de seguimiento
SELECT 
    c.titulo,
    c.estado,
    COUNT(DISTINCT td.id) as tareas_totales,
    COUNT(DISTINCT CASE WHEN td.estado = 'completado' THEN td.id END) as tareas_completadas,
    COUNT(DISTINCT rc.id) as revisiones_pendientes,
    c.fecha_creacion,
    u.nombres as creador_nombre
FROM cursos c
LEFT JOIN tareas_desarrollo td ON c.id = td.curso_id
LEFT JOIN revisiones_cursos rc ON c.id = rc.curso_id AND rc.estado = 'pendiente'
LEFT JOIN usuarios u ON c.creado_por = u.id
GROUP BY c.id, c.titulo, c.estado, c.fecha_creacion, u.nombres;
Vista Diseñador Instruccional:
sql
-- Tareas asignadas por prioridad
SELECT 
    c.titulo as curso,
    m.titulo as modulo,
    td.tipo_contenido,
    td.descripcion_tarea,
    td.fecha_limite,
    td.estado,
    DATEDIFF(td.fecha_limite, NOW()) as dias_restantes
FROM tareas_desarrollo td
JOIN cursos c ON td.curso_id = c.id
LEFT JOIN modulos m ON td.modulo_id = m.id
WHERE td.asignado_a = :user_id
ORDER BY td.fecha_limite ASC;
